      \foreach \i in {1,...,12} {
        % Totally random starting angle for each leg
        \pgfmathsetmacro{\startAngle}{rnd*360}
        \draw[line width=0.2pt, black] (0,0) 
        \foreach \j in {1,...,5} {
          % Short segments (0.1 to 0.25) with sharp random zigzags
          -- ++({\startAngle + (rand*60)}:{0.1 + rnd*0.15})
        };
      }
    \end{scope}
  }

  % --- DRAWING ---

  % . Draw the two blobs
  \drawBlob{L}
  \drawBlob{R}

  % . Draw the radio-style waves

    % The radius grows with each wave
    \pgfmathsetmacro{\currentRadius}{\w * \waveGap}
    % The arc length (angle) increases "crescendo" 
    \pgfmathsetmacro{\arcHalfAngle}{10 + (\w * 6)}
    
    \draw[line width=0.3pt, gray!80] (source) 
      +(\waveDir - \arcHalfAngle:\currentRadius) 
      arc (\waveDir - \arcHalfAngle:\waveDir + \arcHalfAngle:\currentRadius);
  }
